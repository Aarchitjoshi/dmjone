Objective

To understand the implementation of a stack using an array in C++

#include <iostream>
using namespace std;
#define MAX 5

int stack[MAX];
int top = -1;

void push(int val) {
    if(top >= MAX-1) {
        cout << "Stack Overflow" << endl;
    } else {
        stack[++top] = val;
        cout << "Pushed " << val << endl;
    }
}

void pop() {
    if(top < 0) {
        cout << "Stack Underflow" << endl;
    } else {
        cout << "Popped " << stack[top--] << endl;
    }
}

int main() {
    push(5);
    push(4);
    push(3);
    pop();
    pop();

    return 0;
}
