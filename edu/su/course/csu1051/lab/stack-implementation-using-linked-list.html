Objective

To understand the implementation of a stack using linked list in C++

#include <iostream>
using namespace std;

struct Node {
    int data;
    Node* next;
};

class Stack {
private:
    Node* top;
public:
    Stack() { top = NULL; }

    void push(int val) {
        Node* temp = new Node();
        temp->data = val;
        temp->next = top;
        top = temp;
    }

    void pop() {
        if (top == NULL) {
            cout << "Stack is empty" << endl;
            return;
        }
        Node* temp = top;
        top = top->next;
        delete(temp);
    }

    int view() {
        if (top != NULL)
            return top->data;
        else
            return -1;
    }
};

int main() {
    Stack s;
    s.push(1);
    s.push(2);
    s.push(3);
    cout << "Popped: " << s.view() << endl;
    s.pop();
    cout << "Top of Stack: " << s.view() << endl;

    return 0;
}
