Objective

To understand the implementation of a queue using a deque in C++

#include <iostream>
#include <deque>
using namespace std;
int main() {
    // Declare a deque
    deque<int> dq;

    // Insert elements at the rear
    for(int i = 1; i <= 5; i++) {
        dq.push_back(i);
    }

    // Print the queue
    cout << "The queue is: ";
    for(auto i = dq.begin(); i != dq.end(); i++) {
        cout << *i << " ";
    }
    cout << endl;

    // Pop element from the front
    dq.pop_front();

    // Print the queue after popping
    cout << "After popping front, the queue is: ";
    for(auto i = dq.begin(); i != dq.end(); i++) {
        cout << *i << " ";
    }
    cout << endl;

    return 0;
}
